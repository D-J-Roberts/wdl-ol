14 template<typename T>

17 public:
18  IPlugQueue(int size)

20  mData.Resize(size + 1);
21  }

23  ~IPlugQueue(){}

25  bool Push(const T& item)
26  {
27  const auto currentWriteIndex = mWriteIndex.load(std::memory_order_relaxed);
28  const auto nextWriteIndex = Increment(currentWriteIndex);
29  if(nextWriteIndex != mReadIndex.load(std::memory_order_acquire))
30  {
31  mData.Get()[currentWriteIndex] = item;
32  mWriteIndex.store(nextWriteIndex, std::memory_order_release);
33  return true;
34  }
35  return false;
36  }
37 
38  bool Pop(T& item)
39  {
40  const auto currentReadIndex = mReadIndex.load(std::memory_order_relaxed);
41  if(currentReadIndex == mWriteIndex.load(std::memory_order_acquire))
42  {
43  return false; // empty the queue
44  }
45  item = mData.Get()[currentReadIndex];
46  mReadIndex.store(Increment(currentReadIndex), std::memory_order_release);
47  return true;
48  }
49 
50  size_t ElementsAvailable() const
51  {
52  return (mWriteIndex.load(std::memory_order_acquire) - mReadIndex.load(std::memory_order_relaxed))%mData.GetSize();
53  }
54 
55  // useful for reading elements while a criteria is met. Can be used like
56  // while IPlugQueue.ElementsAvailable() && q.peek().mTime < 100 { elem = q.pop() ... }
57  const T& Peek()
58  {
59  const auto currentReadIndex = mReadIndex.load(std::memory_order_relaxed);
60  return mData[currentReadIndex];
61  }
62 
63  bool WasEmpty() const
64  {
65  return (mWriteIndex.load() == mReadIndex.load());
66  }
67 
68  bool WasFull() const
69  {
70  const auto nextWriteIndex = increment(mWriteIndex.load());
71  return (nextWriteIndex == mReadIndex.load());
72  }

74 private:

76  {
77  return (idx + 1) % (mData.GetSize());
78  }
