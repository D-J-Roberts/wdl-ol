54 bool IPlugBase::CompareState(const uint8_t* pIncomingState, int startPos)
55 {
56  bool isEqual = true;
57 
58  const double* data = (const double*) pIncomingState + startPos;
59 
60  // dirty hack here because protools treats param values as 32 bit int and in IPlug they are 64bit float
61  // if we memcmp() the incoming state with the current they may have tiny differences due to the quantization
62  for (int i = 0; i < NParams(); i++)
63  {
64  float v = (float) GetParam(i)->Value();
65  float vi = (float) *(data++);
66 
67  isEqual &= (fabsf(v - vi) < 0.00001);
68  }
69 
70  return isEqual;
71 }
