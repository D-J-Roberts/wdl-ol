169 void invertTransform(float *xform, const float *xformIn)
170 {
171  double d = 1.0 / (xformIn[0] * xformIn[3] - xformIn[1] * xformIn[2]);
172 
173  xform[0] = xformIn[3] * d;
174  xform[1] = -xformIn[2] * d;
175  xform[2] = -xformIn[1] * d;
176  xform[3] = xformIn[0] * d;
177  xform[4] = (-xformIn[4] * xformIn[3] * d) - (xformIn[5] * (-xformIn[2] * d));
178  xform[5] = (-xformIn[4] * (xformIn[1] * d)) - (xformIn[5] * (xformIn[0] * d));
179 }
